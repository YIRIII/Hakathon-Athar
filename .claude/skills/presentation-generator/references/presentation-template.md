# Presentation Template — HTML Structure & Patterns

This template defines the complete structure for self-contained HTML presentations generated by `/presentation-generator`. Every generated presentation must follow this structure exactly.

---

## 1. Document Structure

```html
<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>[Idea Name] — [Audience Type] | Idea Forge</title>
  <style>/* ALL CSS INLINE — see Section 2 */</style>
</head>
<body>
  <div class="presentation" id="presentation">

    <!-- Top-right controls: Language Toggle + Theme Toggle -->
    <div class="top-controls">
      <div class="lang-toggle" id="langToggle">
        <button class="lang-btn active" data-lang="dual">ثنائي | Dual</button>
        <button class="lang-btn" data-lang="ar">عربي</button>
        <button class="lang-btn" data-lang="en">English</button>
      </div>
      <button class="theme-toggle" id="themeToggle" aria-label="Toggle theme" title="Toggle light/dark theme (T)">
        <span class="theme-icon-light">&#9790;</span>
        <span class="theme-icon-dark">&#9788;</span>
      </button>
    </div>

    <!-- Progress Bar -->
    <div class="progress-bar">
      <div class="progress-fill" id="progressFill"></div>
    </div>

    <!-- Slide Counter -->
    <div class="slide-counter" id="slideCounter">1 / N</div>

    <!-- ========== SLIDES ========== -->

    <section class="slide active" data-slide="1">
      <!-- Slide content here -->
      <aside class="speaker-notes">
        <!-- Speaker notes here (hidden by default, toggled with 'S') -->
      </aside>
    </section>

    <!-- ... more slides ... -->

    <!-- ========== NAVIGATION ========== -->
    <nav class="controls">
      <button class="nav-btn prev" id="prevBtn" aria-label="Previous slide">&#8592;</button>
      <button class="nav-btn next" id="nextBtn" aria-label="Next slide">&#8594;</button>
    </nav>

    <!-- Keyboard Shortcuts Help (shown on '?' key) -->
    <div class="shortcuts-overlay" id="shortcutsOverlay">
      <div class="shortcuts-card">
        <h3>Keyboard Shortcuts | اختصارات لوحة المفاتيح</h3>
        <ul>
          <li><kbd>→</kbd> / <kbd>Space</kbd> — Next slide | الشريحة التالية</li>
          <li><kbd>←</kbd> — Previous slide | الشريحة السابقة</li>
          <li><kbd>S</kbd> — Toggle speaker notes | ملاحظات المتحدث</li>
          <li><kbd>P</kbd> — Print mode | وضع الطباعة</li>
          <li><kbd>F</kbd> — Fullscreen | ملء الشاشة</li>
          <li><kbd>T</kbd> — Toggle light/dark theme | تبديل المظهر</li>
          <li><kbd>?</kbd> — This help | هذه المساعدة</li>
          <li><kbd>Esc</kbd> — Close overlay | إغلاق</li>
        </ul>
        <button class="close-btn" onclick="document.getElementById('shortcutsOverlay').classList.remove('visible')">Close | إغلاق</button>
      </div>
    </div>
  </div>

  <script>/* ALL JS INLINE — see Section 4 */</script>
</body>
</html>
```

## 2. CSS Design System

### 2.1 CSS Variables — Light Theme (Default) + Dark Theme

The presentation uses a **light theme by default** for professional settings (officials, partners, customers). A dark theme is available via toggle ('T' key or button). Unless the user specifically requests dark mode as default, always generate with light theme active.

## 4. JavaScript Engine

```javascript
(function() {
  'use strict';

  // === STATE ===
  const slides = document.querySelectorAll('.slide');
  const totalSlides = slides.length;
  let currentSlide = 1;
  let currentLang = 'dual';

  // === NAVIGATION ===
  function goToSlide(n) {
    if (n < 1 || n > totalSlides) return;

    const current = document.querySelector('.slide.active');
    const target = document.querySelector(`[data-slide="${n}"]`);

    if (current) {
      current.classList.remove('active');
      current.classList.add(n > currentSlide ? 'exit-left' : 'exit-right');
      setTimeout(() => {
        current.classList.remove('exit-left', 'exit-right');
      }, 600);
    }

    target.classList.add('active');
    currentSlide = n;
    updateProgress();
    updateCounter();
    triggerAnimations(target);
  }

  function nextSlide() { goToSlide(currentSlide + 1); }
  function prevSlide() { goToSlide(currentSlide - 1); }

  // === PROGRESS ===
  function updateProgress() {
    const pct = ((currentSlide - 1) / (totalSlides - 1)) * 100;
    document.getElementById('progressFill').style.width = pct + '%';
  }

  function updateCounter() {
    document.getElementById('slideCounter').textContent = currentSlide + ' / ' + totalSlides;
  }

  // === ANIMATIONS ===
  function triggerAnimations(slide) {
    const items = slide.querySelectorAll('.animate-in');
    items.forEach(item => {
      item.style.animation = 'none';
      item.offsetHeight; // reflow
      item.style.animation = '';
    });

    // Trigger SVG chart animations
    const bars = slide.querySelectorAll('.chart-bar-fill');
    bars.forEach(bar => {
      const target = bar.getAttribute('data-width');
      bar.style.width = '0';
      setTimeout(() => { bar.style.width = target; }, 100);
    });

    const counters = slide.querySelectorAll('.counter-animate');
    counters.forEach(counter => {
      const target = parseFloat(counter.getAttribute('data-target'));
      const prefix = counter.getAttribute('data-prefix') || '';
      const suffix = counter.getAttribute('data-suffix') || '';
      animateCounter(counter, 0, target, 1500, prefix, suffix);
    });
  }

  function animateCounter(el, start, end, duration, prefix, suffix) {
    const startTime = performance.now();
    function tick(now) {
      const elapsed = now - startTime;
      const progress = Math.min(elapsed / duration, 1);
      const eased = 1 - Math.pow(1 - progress, 3); // ease-out cubic
      const current = start + (end - start) * eased;
      el.textContent = prefix + formatNumber(current) + suffix;
      if (progress < 1) requestAnimationFrame(tick);
    }
    requestAnimationFrame(tick);
  }

  function formatNumber(n) {
    if (n >= 1e9) return (n / 1e9).toFixed(1) + 'B';
    if (n >= 1e6) return (n / 1e6).toFixed(1) + 'M';
    if (n >= 1e3) return (n / 1e3).toFixed(1) + 'K';
    return Math.round(n).toLocaleString();
  }

  // === THEME TOGGLE ===
  function toggleTheme() {
    const current = document.documentElement.getAttribute('data-theme');
    const next = current === 'dark' ? 'light' : 'dark';
    document.documentElement.setAttribute('data-theme', next === 'dark' ? 'dark' : '');
    // Remove attribute entirely for light (default) to use :root vars
    if (next === 'light') document.documentElement.removeAttribute('data-theme');
  }

  document.getElementById('themeToggle').addEventListener('click', toggleTheme);

  // === LANGUAGE TOGGLE ===
  function setLanguage(lang) {
    currentLang = lang;
    document.getElementById('presentation').setAttribute('data-lang', lang);
    document.querySelectorAll('.lang-btn').forEach(btn => {
      btn.classList.toggle('active', btn.getAttribute('data-lang') === lang);
    });
    // Update html dir for single-language modes
    if (lang === 'ar') { document.documentElement.dir = 'rtl'; document.documentElement.lang = 'ar'; }
    else if (lang === 'en') { document.documentElement.dir = 'ltr'; document.documentElement.lang = 'en'; }
    else { document.documentElement.dir = 'rtl'; document.documentElement.lang = 'ar'; }
  }

  document.querySelectorAll('.lang-btn').forEach(btn => {
    btn.addEventListener('click', () => setLanguage(btn.getAttribute('data-lang')));
  });

  // === SPEAKER NOTES ===
  function toggleSpeakerNotes() {
    document.body.classList.toggle('show-notes');
  }

  // === PRINT MODE ===
  function togglePrintMode() {
    document.body.classList.toggle('print-mode');
    if (document.body.classList.contains('print-mode')) {
      // Show all slides
      slides.forEach(s => { s.classList.add('active'); s.style.opacity = '1'; s.style.transform = 'none'; });
    } else {
      // Restore single-slide view
      slides.forEach(s => { s.classList.remove('active'); s.style.opacity = ''; s.style.transform = ''; });
      goToSlide(currentSlide);
    }
  }

  // === FULLSCREEN ===
  function toggleFullscreen() {
    if (!document.fullscreenElement) {
      document.documentElement.requestFullscreen().catch(() => {});
    } else {
      document.exitFullscreen();
    }
  }

  // === KEYBOARD ===
  document.addEventListener('keydown', (e) => {
    if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA') return;

    switch (e.key) {
      case 'ArrowRight':
      case ' ':
        e.preventDefault();
        nextSlide();
        break;
      case 'ArrowLeft':
        e.preventDefault();
        prevSlide();
        break;
      case 's':
      case 'S':
        toggleSpeakerNotes();
        break;
      case 'p':
      case 'P':
        togglePrintMode();
        break;
      case 'f':
      case 'F':
        toggleFullscreen();
        break;
      case 't':
      case 'T':
        toggleTheme();
        break;
      case '?':
        document.getElementById('shortcutsOverlay').classList.toggle('visible');
        break;
      case 'Escape':
        document.getElementById('shortcutsOverlay').classList.remove('visible');
        if (document.body.classList.contains('print-mode')) togglePrintMode();
        break;
    }
  });

  // === NAV BUTTONS ===
  document.getElementById('prevBtn').addEventListener('click', prevSlide);
  document.getElementById('nextBtn').addEventListener('click', nextSlide);

  // === TOUCH SUPPORT ===
  let touchStartX = 0;
  document.addEventListener('touchstart', (e) => { touchStartX = e.touches[0].clientX; });
  document.addEventListener('touchend', (e) => {
    const diff = touchStartX - e.changedTouches[0].clientX;
    if (Math.abs(diff) > 50) {
      if (diff > 0) nextSlide(); else prevSlide();
    }
  });

  // === CITATION CLICKS ===
  document.querySelectorAll('.cite').forEach(cite => {
    cite.addEventListener('click', (e) => {
      e.preventDefault();
      // Navigate to references slide (always the last slide)
      goToSlide(totalSlides);
      // Highlight the specific reference
      const refId = cite.getAttribute('href').replace('#', '');
      const refEl = document.getElementById(refId);
      if (refEl) {
        refEl.style.background = 'var(--accent-glow)';
        setTimeout(() => { refEl.style.background = ''; }, 2000);
      }
    });
  });

  // === INIT ===
  updateProgress();
  updateCounter();
  setLanguage('dual');
  triggerAnimations(document.querySelector('.slide.active'));
})();
```
